        // Procesar Excel
        function procesarExcel(file) {
            const reader = new FileReader();
            reader.onload = function(e) {
                try {
                    const data = new Uint8Array(e.target.result);
                    excelData = XLSX.read(data, {type: 'array'});

                    document.getElementById('excelPreview').classList.add('show');
                    document.getElementById('excelInfo').innerHTML =
                        "<h4>Excel cargado correctamente</h4>" +
                        "<p><strong>Archivo:</strong> " + file.name + "</p>" +
                        "<p><strong>Pestanas:</strong> " + excelData.SheetNames.length + "</p>";

                    const sheetSelect = document.getElementById('sheetSelect');
                    sheetSelect.innerHTML = '';
                    excelData.SheetNames.forEach(name => {
                        const option = document.createElement('option');
                        option.value = name;
                        option.textContent = name;
                        sheetSelect.appendChild(option);
                    });

                    sheetSelect.addEventListener('change', function() {
                        cargarHoja(this.value);
                    });

                    // Buscar pestana de registro hora-hora
                    const hojaPreferida = excelData.SheetNames.find(n =>
                        n.toLowerCase().includes('registro') || n.toLowerCase().includes('hora')
                    ) || excelData.SheetNames[0];

                    sheetSelect.value = hojaPreferida;
                    cargarHoja(hojaPreferida);

                } catch (err) {
                    alert("Error al leer Excel: " + err.message);
                }
            };
            reader.onerror = function() {
                alert("Error al cargar el archivo");
            };
            reader.readAsArrayBuffer(file);
        }

            if (jsonData.length < 2) {
                alert('La hoja esta vacia');
                return;
            }

            // Buscar fila de cabeceras - formato ARZYZ: Fila 2 = cabeceras
            // Estructura: Fila0=titulo, Fila1=vacia, Fila2=cabeceras
            let headerRow = 2; // Por defecto buscar en fila 2 para ARZYZ
            let detectarFilaCorrecta = false;

            // Primero verificar si fila 2 tiene las cabeceras correctas
            if (jsonData[2] && jsonData[2].length > 0) {
                const row2Str = String(jsonData[2]).toLowerCase();
                if (row2Str.includes('fecha') && row2Str.includes('hora')) {
                    headerRow = 2;
                    detectarFilaCorrecta = true;
                    console.log("Cabeceras detectadas en fila 2");
                }
            }

            // Si no, buscar automaticamente en las primeras 10 filas
            if (!detectarFilaCorrecta) {
                for (let i = 0; i < Math.min(10, jsonData.length); i++) {
                    const row = jsonData[i];
                    if (!row || !row.length || row.length < 2) continue;

                    // Contar cuantas columnas tienen palabras clave
                    let score = 0;
                    const rowStr = String(row).toLowerCase();
                    if (rowStr.includes('fecha')) score++;
                    if (rowStr.includes('hora')) score++;
                    if (rowStr.includes('t') && (rowStr.includes('ºc') || rowStr.includes('°c') || rowStr.includes('temp'))) score++;

                    // Si tiene al menos 2 indicadores, es la fila de cabeceras
                    if (score >= 2) {
                        headerRow = i;
                        console.log("Cabeceras detectadas en fila " + i);
                        break;
                    }
                }
            }

            const headers = jsonData[headerRow] || [];

            
            
            console.log("Filas totales:", jsonData.length);
            console.log("Fila de cabeceras:", headerRow);
            console.log("Cabeceras (raw):", JSON.stringify(headers));

                    const n = String(t.nombre).toLowerCase();
                    // Detectar cualquier Q + número

                            const excelEpoch = new Date(1899, 11, 30);

        // Procesar datos - NUEVA VERSION: Calcula k para cada termopar
        function procesarDatos() {
            const jsonData = XLSX.utils.sheet_to_json(currentSheet, {header: 1});
            if (jsonData.length < 3) return [];

            // Buscar fila de cabeceras
            let headerRow = 0;
            for (let i = 0; i < Math.min(10, jsonData.length); i++) {
                const row = jsonData[i];
                if (!row) continue;
                const rowStr = JSON.stringify(row).toLowerCase();
                if (rowStr.includes('fecha') || rowStr.includes('hora') || rowStr.includes('tª') || rowStr.includes('temp')) {
                    headerRow = i;
                    break;
                }
            }

            const headers = jsonData[headerRow] || [];

            // Identificar columnas de temperatura
            const colTemperatura = {}; // {nombre: indice}
            let colAislante = -1;
            let colRampa = -1;

            headers.forEach((h, idx) => {
                if (!h) return;
                const hLower = String(h).toLowerCase().replace(/ºc|\(ºc\)|°c/g, '').trim();
                if (hLower.includes('aislante')) {
                    colAislante = idx;
                } else if (hLower.includes('rampa')) {
                    colRampa = idx;
                } else if (hLower.includes('t') && !hLower.includes('nave')) {
                    // Termopar de temperatura (Q1, Q2, etc, excluyendo NAVES)
                    colTemperatura[h] = idx;
                }
            });

            // Datos de la pieza (ya convertidos en calcularK, aqui solo se usan)
            // Obtener datos según el modo
            const mismoHormigon = document.getElementById('mismoHormigon').value;

            // Función para obtener datos de un termopar específico
            function getDatosTermopar(termoparIdx) {
                if (mismoHormigon === 'si') {
                    return {
                        masa: parseNum(document.getElementById('masa').value),
                        espesor: parseNum(document.getElementById('espesor').value) / 1000,
                        area: parseNum(document.getElementById('areaCalculada').textContent)
                    };
                } else {
                    return {
                        masa: parseNum(document.getElementById('masa_tp_' + termoparIdx).value),
                        espesor: parseNum(document.getElementById('espesor_tp_' + termoparIdx).value) / 1000,
                        base: parseNum(document.getElementById('base_tp_' + termoparIdx).value),
                        altura: parseNum(document.getElementById('altura_tp_' + termoparIdx).value)
                    };
                }
            }

            // Datos globales (para validación)
            const masaGlobal = parseNum(document.getElementById('masa').value);
            const espesorMm = parseNum(document.getElementById('espesor').value);
            const espesorGlobal = espesorMm / 1000;
            const areaGlobal = parseNum(document.getElementById('areaCalculada').textContent);

            // Obtener termopares Q (cara caliente) para el cálculo
            // Filtrar solo Q + número de thermoparesDetectados
            const termoparesQ = thermoparesDetectados.filter(tp => {
                return /q\d+/i.test(String(tp.nombre));
            });

            const termoparesCalculo = termoparesQ.map((tp, idx) => {
                // Obtener índice del hormigón seleccionado (el select usa idx del array filtrado)
                const selectHormigon = document.getElementById('hormigon_tp_' + termoparIdx);
                let hormigon;
                if (selectHormigon.value === 'otro') {
                    const nombreOtro = document.getElementById('hormigon_otro_' + termoparIdx).value || 'Otro';
                    const cpOtro = parseFloat(document.getElementById('cp_otro_' + termoparIdx).value) || 0.96;
                    hormigon = { name: nombreOtro, cp: cpOtro };
                } else {
                    hormigon = HormigonesDatabase[parseInt(selectHormigon.value)] || HormigonesDatabase[0];
                }

                return {
                    nombre: tp.nombre,
                    columna: tp.columna,
                    hormigon: hormigon.name,
                    cp: hormigon.cp
                };
            });

            if (colAislante < 0 || termoparesCalculo.length === 0) {
                alert('No se pudo identificar la columna AISLANTE o termopares de cara caliente');
                return [];
            }

            // Identificar columna de Nº HORA (para diferencia de horas)
            let colNumHora = -1;
            headers.forEach((h, idx) => {
                if (!h) return;
                const hLower = String(h).toLowerCase();
                if (hLower.includes('nº hora') || hLower.includes('numero hora') || hLower.includes('n° hora')) {
                    colNumHora = idx;
                }
            });

            // Identificar mesetas (basado en la temperatura de rampa o el primer termopar)
            const mesetas = [];
            let tempActual = null;
            let filaInicioMeseta = null;
            let consecutiveCount = 0;

            for (let i = headerRow + 1; i < jsonData.length; i++) {
                const row = jsonData[i];
                const tempRampa = colRampa >= 0 ? parseFloat(row ? row[colRampa] : null) : null;

                if (tempRampa !== null && !isNaN(tempRampa)) {
                    if (tempRampa === tempActual) {
                        consecutiveCount++;
                    } else {
                        if (tempActual !== null && consecutiveCount >= 2) {
                            mesetas.push({temp: tempActual, inicio: filaInicioMeseta, fin: i - 1});
                        }
                        tempActual = tempRampa;
                        filaInicioMeseta = i;
                        consecutiveCount = 1;
                    }
                }
            }

            if (tempActual !== null && consecutiveCount >= 2) {
                mesetas.push({temp: tempActual, inicio: filaInicioMeseta, fin: jsonData.length - 1});
            }

            // Calcular k para cada termopar y cada meseta
            const resultados = []; // {termopar, hormigon, meseta, temp, tCaliente, tFrio, deltaT, k, Q}

            termoparesCalculo.forEach(termopar => {
                mesetas.forEach((meseta, mesetaIdx) => {
                    // Verificar que hay una fila antes de la meseta para calcular dT/dt
                    if (meseta.inicio <= 0) return;

                    // Obtener temperatura DEL AISLANTE antes de la meseta (fila anterior)
                    const rowAntes = jsonData[meseta.inicio - 1];
                    const tAislanteAntes = colAislante >= 0 ? parseFloat(rowAntes ? rowAntes[colAislante] : null) : null;

                    // Obtener temperatura DEL AISLANTE al final de la meseta (penúltima fila para evitar que cambie)
                    const rowFin = jsonData[meseta.fin - 1] || jsonData[meseta.fin];
                    const tAislanteFin = colAislante >= 0 ? parseFloat(rowFin ? rowFin[colAislante] : null) : null;

                    // Calcular deltaTFrio: T_fin - T_antes (no durante la meseta)
                    if (tAislanteAntes === null || tAislanteFin === null || isNaN(tAislanteAntes) || isNaN(tAislanteFin)) {
                        return;
                    }
                    const deltaTFrio = tAislanteFin - tAislanteAntes;

                    // Calcular deltaHoras: hora fin - hora antes (usar las mismas filas que el aislante)
                    let deltaHoras = 1;
                    if (colNumHora >= 0) {
                        const horaAntes = parseFloat(rowAntes ? rowAntes[colNumHora] : null);
                        const horaFin = meseta.fin > 0 ? parseFloat((jsonData[meseta.fin - 1] || jsonData[meseta.fin]) ? (jsonData[meseta.fin - 1] || jsonData[meseta.fin])[colNumHora] : null) : null;
                        if (!isNaN(horaAntes) && !isNaN(horaFin) && horaFin > horaAntes) {
                            deltaHoras = horaFin - horaAntes;
                        }
                    }

                    // Calcular PROMEDIO de todos los valores durante la meseta
                    let sumCaliente = 0, sumFrio = 0, count = 0;

                    for (let i = meseta.inicio; i <= meseta.fin; i++) {
                        const row = jsonData[i];
                        if (!row) continue;

                        // Temperatura del termopar (cara caliente) - PROMEDIO
                        const tCaliente = parseFloat(row[termopar.columna]);
                        if (!isNaN(tCaliente)) {
                            sumCaliente += tCaliente;
                        }

                        // Temperatura del aislante (cara fria) - PROMEDIO
                        if (colAislante >= 0) {
                            const tFrio = parseFloat(row[colAislante]);
                            if (!isNaN(tFrio)) {
                                sumFrio += tFrio;
                            }
                            count++;
                        }
                    }

                    if (count === 0) return;

                    // PROMEDIO durante la meseta (no solo inicio y fin)
                    const tCalienteMedia = sumCaliente / count;
                    const tFrioMedia = sumFrio / count;
                    const deltaT = tCalienteMedia - tFrioMedia;

                    if (deltaHoras <= 0 || deltaT <= 0) return;

                    const dT_dt = deltaTFrio / deltaHoras; // K/hora (REAL)

                    // k = (m * Cp * 1000 * dT/dt * e) / (A * ΔT_caras)
                    // dT/dt ya está en K/hora, convertir a K/s: /3600
                    const k = (masa * termopar.cp * 1000 * (dT_dt / 3600) * espesor) / (area * deltaT);
                    const Q = k * area * deltaT / espesor;

                    resultados.push({
                        termopar: termopar.nombre,
                        hormigon: termopar.hormigon,
                        cp: termopar.cp,
                        meseta: mesetaIdx + 1,
                        temp: meseta.temp,
                        tCaliente: tCalienteMedia,
                        tFrio: tFrioMedia,
                        deltaT: deltaT,
                        deltaTFrio: deltaTFrio,
                        deltaHoras: deltaHoras,
                        dT_dt: dT_dt,
                        k: k,
                        Q: Q
                    });
                });
            });

            return resultados;
        }

        function mostrarGrafico(resultados) {
            const ctx = document.getElementById('kChart').getContext('2d');

            if (chartInstance) {
                chartInstance.destroy();
            }

            // Agrupar por termopar
            const termopares = [...new Set(resultados.map(r => r.termopar))];
            const tempsUnicas = [...new Set(resultados.map(r => r.temp))].sort((a,b) => a-b);
            const labels = tempsUnicas.map(t => t + '°C');

            const colors = ['#E31837', '#2196F3', '#4CAF50', '#FF9800', '#9C27B0', '#00BCD4', '#795548'];

            const datasets = termopares.map((tp, idx) => {
                const data = tempsUnicas.map(temp => {
                    const r = resultados.find(r => r.termopar === tp && r.temp === temp);
                    return r ? r.k : null;
                });
                return {
                    label: tp,
                    data: data,
                    borderColor: colors[idx % colors.length],
                    backgroundColor: colors[idx % colors.length] + '20',
                    fill: false,
                    tension: 0.3,
                    pointRadius: 5,
                    pointBackgroundColor: colors[idx % colors.length]
                };
            });

            chartInstance = new Chart(ctx, {
                type: 'line',
                data: { labels: labels, datasets: datasets },
                options: {
                    responsive: true,
                    plugins: { legend: { position: 'top' } },
                    scales: {
                        y: { beginAtZero: false, title: { display: true, text: 'Conductividad k (W/(m·K))' } },
                        x: { title: { display: true, text: 'Temperatura Meseta (°C)' } }
                    }
                }
            });
        }

        function mostrarComparacion(resultados) {
            const container = document.getElementById('comparisonGrid');
            container.innerHTML = '';

            // Obtener lista de hormigones usados
            const hormigonesUsados = [...new Set(resultados.map(r => r.hormigon))];

            if (hormigonesUsados.length === 0) {
                container.innerHTML = '<p>No hay datos para comparar</p>';
                return;
            }

            // Por cada hormigón, mostrar comparación
            hormigonesUsados.forEach(hormigon => {
                const hData = HormigonesDatabase.find(h => h.name === hormigon);
                if (!hData || !hData.k) return;

                const resultadosHormigon = resultados.filter(r => r.hormigon === hormigon);
                const tempsMesetas = [...new Set(resultadosHormigon.map(r => r.temp))];
                
                // Obtener las temperaturas de la ficha técnica
                const tempsK = Object.keys(hData.k).map(Number).sort((a,b) => a-b);
                
                // Encontrar la meseta cuya temperatura sea más cercana a alguna temperatura de la ficha
                let mejorTemp = null;
                let mejorDiff = Infinity;
                let tempFichaCercana = null;
                
                tempsMesetas.forEach(tempMeseta => {
                    tempsK.forEach(tempK => {
                        const diff = Math.abs(tempMeseta - tempK);
                        if (diff < mejorDiff) {
                            mejorDiff = diff;
                            mejorTemp = tempMeseta;
                            tempFichaCercana = tempK;
                        }
                    });
                });

                // Calcular la MEDIA de k de todos los termopares en esa meseta
                const resultadosMejorMeseta = resultadosHormigon.filter(r => r.temp === mejorTemp);
                const kMedia = resultadosMejorMeseta.reduce((sum, r) => sum + r.k, 0) / resultadosMejorMeseta.length;
                const termoparesEnMeseta = resultadosMejorMeseta.map(r => r.termopar).join(', ');

                const k = hData.k[tempFichaCercana];
                const diff = ((kMedia - k) / k * 100);
                const diffClass = Math.abs(diff) < 15 ? 'positive' : (Math.abs(diff) < 30 ? 'warning' : 'negative');

                container.innerHTML +=
                    '<div class="comparison-section">' +
                    '<h4>' + hormigon + '</h4>' +
                    '<div class="comparison-item"><p>Termopares: ' + termoparesEnMeseta + '</p><p>Temp meseta: ' + mejorTemp + '°C</p><p>Temp ficha: ' + tempFichaCercana + '°C</p>' +
                    '<div class="comparison-item"><p>k calculada (media): ' + kMedia.toFixed(3) + ' W/(m·K)</strong></p><p>k ficha: <strong>' + k.toFixed(2) + ' W/(m·K)</strong></p>' +
                    '<div class="comparison-item"><div class="diff ' + diffClass + '">Diferencia: ' + diff.toFixed(1) + '%' +
                    '';
            });
        }

        function mostrarConclusiones(resultados) {
            const container = document.getElementById('conclusionsList');
            const proyecto = document.getElementById('proyecto').value;
            let html = '';

            if (resultados.length === 0) {
                html = '<div class="conclusion-item">No se detectaron mesetas en los datos';
            } else {
                // Información por termopar
                const termopares = [...new Set(resultados.map(r => r.termopar))];
                const tempsUnicas = [...new Set(resultados.map(r => r.temp))];
                const numMesetas = tempsUnicas.length;
                const numTermopares = termopares.length;

                html += '<div class="conclusion-item"><strong>Proyecto:</strong> ' + proyecto + '';
                html += '<div class="conclusion-item"><strong>Mesetas detectadas:</strong> ' + numMesetas + '';
                html += '<div class="conclusion-item"><strong>Termopares por meseta:</strong> ' + numTermopares + '';
                html += '<div class="conclusion-item"><strong>Termopares:</strong> ' + termopares.join(', ') + '';

                // Información por hormigón
                const hormigones = [...new Set(resultados.map(r => r.hormigon))];
                html += '<div class="conclusion-item"><strong>Hormigones:</strong> ' + hormigones.join(', ') + '';

                // Resumen por hormigón
                hormigones.forEach(h => {
                    const resH = resultados.filter(r => r.hormigon === h);
                    const kMax = Math.max(...resH.map(r => r.k));
                    const kMin = Math.min(...resH.map(r => r.k));
                    const kAvg = resH.reduce((sum, r) => sum + r.k, 0) / resH.length;
                    html += '<div class="conclusion-item"><strong>' + h + ':</strong> k media = ' + kAvg.toFixed(4) + ' W/(m·K) (rango: ' + kMin.toFixed(3) + ' - ' + kMax.toFixed(3) + ')';
                });

                if (resultados.length > 1) {
                    const trend = resultados[resultados.length - 1].k > resultados[0].k;
                    html += '<div class="conclusion-item"><strong>Observacion:</strong> La conductividad ' + (trend ? 'aumenta' : 'disminuye') + ' con la temperatura';
                }

                const tempMax = Math.max(...resultados.map(r => r.temp));
                if (tempMax < 400) {
                    html += '<div class="conclusion-item"><strong>Nota:</strong> Temperaturas bajas pueden tener agua residual. Precision limitada.';
                } else {
                    html += '<div class="conclusion-item">Temperaturas >400C dan mejor estimacion de k real.';
                }
            }

            container.innerHTML = html;
        }
